def isVowel(c):
    return "AEIOUÁÉÍÓÚÀÈÌÒÙ".find(c.upper()) != -1

def divide(word):
    a = word.upper()
    result = ""
    s = 0
    l = len(a)
    while s < l:
        if s+1 == l:
            result += word[s]; s += 1
        elif not isVowel(a[s]):
            result += word[s]; s += 1
        elif not isVowel(a[s+1]):
            if s+2 >= len(a):
                result += word[s:s+2]+"-"; s += 2
            elif isVowel(a[s+2]):
                result += word[s]+"-"; s += 1
            elif a[s+1] == a[s+2]:
                result += word[s:s+2]+"-"; s += 2
            elif a[s+1] in "SG":
                result += word[s]+"-"; s += 1
            elif a[s+2] in "RLH":
                result += word[s]+"-"; s += 1
            else:
                result += word[s:s+2]+"-"; s += 2
        elif a[s+1] in "IÍÌ":
            if s+2<=l:
                result += word[s]; s += 1
            elif s>1 and a[s-1:s+1]=="QU" and isVowel(a[s+2]):
                result += word[s:s+2]; s += 2
            elif isVowel(a[s+2]):
                result += word[s]+"-"; s += 1
            else:
                result += word[s]; s += 1
        elif a[s] in "IÍÌUÚÙ":
            result += word[s]; s += 1
        else:
            result += word[s]+"-"; s += 1
    
    if result[-1] == "-":
        result = result[:-1]
    return result

def lego_sillabe(sillabe, dizionario):
    regex_str = r"^(" + "|".join(sillabe)+ r")+$"
    #print(regex_str)
    regex = re.compile(regex_str)
    found = []
    for iw in dizionario:
        if regex.match(iw.rstrip()):
            found.append(iw)
    return found
